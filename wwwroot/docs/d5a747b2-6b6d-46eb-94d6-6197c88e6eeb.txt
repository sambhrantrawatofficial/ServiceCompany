@section Scripts {
    <script>
        // Add extra address field
        function addAddress() {
            var container = document.getElementById("address-container");
            var input = document.createElement("input");
            input.type = "text";
            input.name = "addresses";
            input.className = "form-control mb-2";
            container.appendChild(input);
        }
        // Add extra document upload field
        function addDoc() {
            var container = document.getElementById("docs-container");
            var input = document.createElement("input");
            input.type = "file";
            input.name = "verificationDocs";
            input.className = "form-control mb-2";
            container.appendChild(input);
        }
        // Drag & Drop logic for all draggable items
        document.querySelectorAll('.draggable').forEach(item => {
            item.addEventListener('dragstart', e => {
                e.dataTransfer.setData('text', e.target.innerText);
            });
        });
        // Weekly schedule drop
        let scheduleDrop = document.getElementById("schedule-dropzone");
        scheduleDrop.addEventListener('dragover', e => e.preventDefault());
        scheduleDrop.addEventListener('drop', e => {
            e.preventDefault();
            let day = e.dataTransfer.getData('text');
            scheduleDrop.innerHTML += `<span class="badge bg-success m-1">${day}</span>`;
            document.getElementById("WeeklySchedule").value += day + "|";
        });
        // Pin code drop
        let pinDrop = document.getElementById("pin-dropzone");
        pinDrop.addEventListener('dragover', e => e.preventDefault());
        pinDrop.addEventListener('drop', e => {
            e.preventDefault();
            let num = e.dataTransfer.getData('text');
            pinDrop.innerHTML += `<span class="badge bg-dark text-white m-1">${num}</span>`;
            document.getElementById("PinCode").value += num;
        });

        // Radius slider
        let radiusRange = document.getElementById("radiusRange");
        let radiusValue = document.getElementById("radiusValue");
        radiusRange.addEventListener("input", function () {
            radiusValue.innerText = this.value + " km";
            document.getElementById("Radius").value = this.value;
        });

        // Time slots drop
        let slotsDrop = document.getElementById("slots-dropzone");
        slotsDrop.addEventListener('dragover', e => e.preventDefault());
        slotsDrop.addEventListener('drop', e => {
            e.preventDefault();
            let slot = e.dataTransfer.getData('text');
            slotsDrop.innerHTML += `<span class="badge bg-warning text-dark m-1">${slot}</span>`;
            document.getElementById("TimeSlots").value += slot + "|";
        });
    </script>
}
